Review Sesi 3 – Pewarisan

  • Inheritance: extends keyword
  • super() untuk panggil constructor parent
  • @override untuk override method
  • Contoh cepat:
class Hewan {
  void bersuara() => print('...');
}
class Kucing extends Hewan {
  @override
  void bersuara() => print('Meong');
}

Pengantar Polimorfisme

  • Arti: "Banyak bentuk" (poly = banyak, morph = bentuk)( )
  • Konsep: Objek dari subclass dapat diperlakukan sebagai objek dari superclass( )
  • Contoh: Semua Kucing , Anjing , Burung adalah Hewan
  • Tipe: i. Compile-time polymorphism: Method overloading (tidak didukung Dart) ii. Runtime polymorphism: Method overriding (didukung Dart)( )

Polimorfisme dengan Inheritance

  • Referensi tipe parent dapat menampung objek child
class Bentuk {
  void gambar() => print('Menggambar bentuk');
}
class Lingkaran extends Bentuk {
  @override
  void gambar() => print('Menggambar lingkaran');
}
class Persegi extends Bentuk {
  @override
  void gambar() => print('Menggambar persegi');
}
void main() {
  Bentuk bentuk1 = Lingkaran();  // Polimorfisme
  Bentuk bentuk2 = Persegi();    // Polimorfisme
  bentuk1.gambar();  // Output: Menggambar lingkaran
  bentuk2.gambar();  // Output: Menggambar persegi
}

Manfaat Polimorfisme

  • Fleksibilitas: Kode dapat bekerja dengan berbagai tipe
  • Extensibility: Mudah menambah tipe baru
  • Code Reusability: Method umum di superclass
  • Maintainability: Perubahan mudah dikelola

is dan as Operator

  • is : Mengecek tipe objek
  • as : Casting tipe objek
class Kendaraan {}
class Mobil extends Kendaraan {
  void klakson() => print('Tin!');
}
void main() {
  Kendaraan k = Mobil();
  if (k is Mobil) {          
  // Cek tipe
  (k as Mobil).klakson();  // Casting
  }
  // Safe casting dengan as?
  (k as Mobil?)?.klakson();
}

Pengantar Abstraksi

  • Konsep: Menyembunyikan detail implementasi, hanya menunjukkan fungsionalitas
  • Tujuan: i. Fokus pada "apa" bukan "bagaimana" ii. Kontrak yang harus dipenuhi iii. Fleksibilitas implementasi
  • Cara di Dart: abstract class dan interface

Abstract Class

  • Tidak bisa diinstansiasi langsung
  • Dapat memiliki method abstract (tanpa implementasi) dan concrete (dengan implementasi)
abstract class Kendaraan {
  String? merk;  // Field concrete
  
  void start();  // Method abstract
  
  void stop() {  // Method concrete
    print('Kendaraan berhenti');
  }
}
class Mobil extends Kendaraan {
  @override
  void start() {
    print('Mobil $merk menyala');
  }
}

Abstract Method

  • Method tanpa body (implementasi)
  • Harus di-override di subclass
abstract class AlatMusik {
  void mainkan();  // Abstract method
  
  void info() {    // Concrete method
  print('Ini adalah alat musik');
  }
}
  
class Gitar extends AlatMusik {
  @override
  void mainkan() {
    print('Jreng jreng...');
  }
}

Interface di Dart

  • Fakta: sejak Dart 3.0 dart punya keyword interface
  • Cara: Semua interface class adalah implicit interface
  • Gunakan implements untuk mengimplementasi interface
interface class Flyable {
  void terbang() => print('Terbang...');
}
  
class Bird implements Flyable {
  @override
  void terbang() => print('Burung terbang tinggi');
}
  
  class Plane implements Flyable {
    @override
    void terbang() => print('Pesawat terbang cepat');
}

class User {

  String _username;
  String _password;
  String _email;

  User(this._username, this._password, this._email);

  // Getter
  String get username => _username;
  String get password => _password;
  String get email => _email;

  // Setter Username
  set username(String value) {
    if (value.length >= 5) {
      _username = value;
      print("Username berhasil diubah menjadi: $_username");
    } else {
      print("Username gagal diubah. Minimal 5 karakter.");
    }
  }

  // Setter Password
  set password(String value) {
    if (value.length >= 8) {
      _password = value;
      print("Password berhasil diubah.");
    } else {
      print("Password gagal diubah. Minimal 8 karakter.");
    }
  }

  // Setter Email
  set email(String value) {
    if (value.contains("@")) {
      _email = value;
      print("Email berhasil diubah menjadi: $_email");
    } else {
      print("Email tidak valid. Harus mengandung '@'.");
    }
  }
}

void main() {

  // Membuat objek
  User user1 = User("bagas123", "password123", "bagas@email.com");

  // Menampilkan data awal
  print("Username: ${user1.username}");
  print("Password: ${user1.password}");
  print("Email: ${user1.email}");

  print("");

  // Mencoba mengubah data
  user1.username = "bag";        // gagal
  user1.username = "bagas_k";    // berhasil

  user1.password = "123";        // gagal
  user1.password = "secure123";  // berhasil

  user1.email = "bagasemail";    // gagal
  user1.email = "bagas@mail.com";// berhasil

}

class User {

  User(this._username, this._password);
  String get username => _username;
  bool login(String username, String password) {
    return _username == username && _password == password;
  }
  
  void gantiPassword(String lama, String baru) {
    if (_password == lama) { // Now _password refers to the instance variable of this User object
      if (baru.length >= 8) {
        _password = baru;
        print('Password berhasil diubah.');
      } else {
        print('Gagal mengubah password: Password baru harus memiliki minimal 8 karakter.');
      }
    } else {
      print('Gagal mengubah password: Password lama tidak sesuai.');
    }
  }
}

class RekeningBank {

  String _nomorRekening;
  double _saldo;

  // Constructor
  RekeningBank(this._nomorRekening, this._saldo);

  // Method untuk menambah saldo
  void deposit(double jumlah) {
    if (jumlah > 0) {
      _saldo += jumlah;
      print("Deposit berhasil: Rp$jumlah");
    } else {
      print("Jumlah deposit tidak valid.");
    }
  }

  // Method untuk menarik saldo
  void tarik(double jumlah) {
    if (jumlah <= 0) {
      print("Jumlah penarikan tidak valid.");
    } else if (jumlah > _saldo) {
      print("Saldo tidak mencukupi.");
    } else {
      _saldo -= jumlah;
      print("Penarikan berhasil: Rp$jumlah");
    }
  }

  // Method untuk mengecek saldo
  void cekSaldo() {
    print("Nomor Rekening : $_nomorRekening");
    print("Saldo Saat Ini : Rp$_saldo");
  }
}

void main() {

  // Membuat objek rekening
  RekeningBank rekening1 = RekeningBank("251410016", 900000);

  print("=== Saldo Awal ===");
  rekening1.cekSaldo();

  print("\n=== Deposit ===");
  rekening1.deposit(200000);

  print("\n=== Tarik Saldo ===");
  rekening1.tarik(100000);

  print("\n=== Coba Tarik Melebihi Saldo ===");
  rekening1.tarik(1000000);

  print("\n=== Saldo Akhir ===");
  rekening1.cekSaldo();

}
