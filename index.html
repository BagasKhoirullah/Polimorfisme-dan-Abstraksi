  • Contoh cepat:
class Hewan {
  void bersuara() => print('...');
}

  @override
  void bersuara() => print('Meong');

Pengantar Polimorfisme

Arti: "Banyak bentuk" (poly = banyak, morph = bentuk)

Konsep: Objek dari subclass dapat diperlakukan sebagai objek dari superclass.

Contoh: Kucing, Anjing, dan Burung adalah Hewan.

  • Compile-time polymorphism: Method overloading (tidak didukung Dart)
  • Runtime polymorphism: Method overriding (didukung Dart)

Polimorfisme dengan Inheritance

  • Referensi tipe parent dapat menampung objek child
class Bentuk {
  void gambar() => print('Menggambar bentuk');
}
class Lingkaran extends Bentuk {
  @override
  void gambar() => print('Menggambar lingkaran');
}
class Persegi extends Bentuk {
  @override
  void gambar() => print('Menggambar persegi');
}
void main() {
  Bentuk bentuk1 = Lingkaran();  // Polimorfisme
  Bentuk bentuk2 = Persegi();    // Polimorfisme
  bentuk1.gambar();  // Output: Menggambar lingkaran
  bentuk2.gambar();  // Output: Menggambar persegi
}

Manfaat Polimorfisme

  • Fleksibilitas: Kode dapat bekerja dengan berbagai tipe
  • Extensibility: Mudah menambah tipe baru
  • Code Reusability: Method umum di superclass
  • Maintainability: Perubahan mudah dikelola

is dan as Operator

  • is : Mengecek tipe objek
  • as : Casting tipe objek
class Kendaraan {}
class Mobil extends Kendaraan {
  void klakson() => print('Tin!');
}

void main() {
  Kendaraan k = Mobil();
  if (k is Mobil) {           // Cek tipe
    (k as Mobil).klakson();  // Casting
  }
  
  // Safe casting dengan as?
  (k as Mobil?)?.klakson();
}

Praktik 1 – Polimorfisme dengan List

  • Buat list List<Hewan>
  • Tambahkan berbagai objek subclass (Kucing , Anjing , Burung )
  • Loop list, panggil method bersuara()
class Hewan {
  void bersuara() {
    print("Hewan bersuara");
  }
}

class Kucing extends Hewan {
  @override
  void bersuara() {
    print("Meong");
  }
}

class Anjing extends Hewan {
  @override
  void bersuara() {
    print("Guk guk");
  }
}

class Burung extends Hewan {
  @override
  void bersuara() {
    print("Cuit cuit");
  }
}

void main() {
  // List bertipe Hewan (polimorfisme)
  List daftarHewan = [
    Kucing(),
    Anjing(),
    Burung()
  ];

  // Loop dan panggil method
  for (var hewan in daftarHewan) {
    hewan.bersuara();
  }
}

Pengantar Abstraksi

Konsep: Menyembunyikan detail implementasi, hanya menunjukkan fungsionalitas.

Tujuan:

  • Fokus pada "apa" bukan "bagaimana"
  • Menjadi kontrak yang harus dipenuhi
  • Memberikan fleksibilitas implementasi

Cara di Dart: menggunakan abstract class dan interface

Abstract Class

  • Tidak bisa diinstansiasi langsung
  • Dapat memiliki method abstract (tanpa implementasi) dan concrete (dengan implementasi)
abstract class Kendaraan {
  String? merk;  // Field concrete
  
  void start();  // Method abstract
  
  void stop() {  // Method concrete
    print('Kendaraan berhenti');
  }
}
class Mobil extends Kendaraan {
  @override
  void start() {
    print('Mobil $merk menyala');
  }
}

Abstract Method

  • Method tanpa body (implementasi)
  • Harus di-override di subclass
abstract class AlatMusik {
  void mainkan();  // Abstract method
  
  void info() {    // Concrete method
  print('Ini adalah alat musik');
  }
}
  
class Gitar extends AlatMusik {
  @override
  void mainkan() {
    print('Jreng jreng...');
  }
}

Interface di Dart

  • Fakta: sejak Dart 3.0 dart punya keyword interface
  • Cara: Semua interface class adalah implicit interface
  • Gunakan implements untuk mengimplementasi interface
interface class Flyable {
  void terbang() => print('Terbang...');
}
  
class Bird implements Flyable {
  @override
  void terbang() => print('Burung terbang tinggi');
}
  
  class Plane implements Flyable {
    @override
    void terbang() => print('Pesawat terbang cepat');
}

Perbedaan extends vs implements

Aspect extends implements
Hubungan "is-a" "can-do"
Inheritance Mewarisi implementasi Hanya kontrak
Method Bisa concrete/abstract Semua harus di-implement
Inheritance Type Single inheritance Multiple inheritance

Static Members dan Method

  • Static members adalah properti atau method yang dimiliki oleh class itu sendiri, bukan oleh objek class. Static member diakses langsung melalui class name.
  • Properti dan method static sangat berguna untuk membuat utility class atau menyimpan konstanta yang terkait dengan class, sehingga dapat diakses langsung tanpa perlu membuat objek.
class MathUtils {
  static double pi = 3.14;
  static int add(int x, int y) {
    return x + y;
  }
}
  
void main() {
  print('Nilai pi : ${MathUtils.pi}');
  print('Result : ${MathUtils.add(10, 20)}');
}

Studi Kasus: Sistem Bentuk Geometri

abstract class Bentuk {
  double hitungLuas();
  double hitungKeliling();
  
  void info() {
    print('Luas: ${hitungLuas()}, Keliling: ${hitungKeliling()}');
  }
}
  
class Persegi implements Bentuk {
  double sisi;
  Persegi(this.sisi);
  
  @override
  double hitungLuas() => sisi * sisi;
  
  @override
  double hitungKeliling() => 4 * sisi;
  
  @override
  void info() {
    print('Persegi dengan sisi ${sisi} memiliki luas ${hitungLuas()} dan keliling ${hitungKeliling()}.');
  }
}
  
class Lingkaran implements Bentuk {
  double radius;
  Lingkaran(this.radius);
  
  @override
  double hitungLuas() => 3.14 * radius * radius;
  
  @override
  double hitungKeliling() => 2 * 3.14 * radius;
}
  
@override
void info() {
print('Lingkaran dengan radius ${radius} memiliki luas ${hitungLuas()} dan keliling ${hitungKeliling()}.');
}

Ringkasan

  • Polimorfisme: Objek child dapat diperlakukan sebagai parent
  • Abstract class: Template dengan method abstract/concrete
  • Interface: Kontrak yang harus diimplementasi
  • Multiple inheritance: mewarisi fungsi dari dua atau lebih hierarki class yang berbeda

// 5. MathUtils
import 'dart:math';

class MathUtils {
  static int factorial(int n) => n <= 1 ? 1 : n * factorial(n-1);

  static bool isPrime(int n) {
    if (n < 2) return false;
    for (int i=2;i<=sqrt(n);i++){
      if(n%i==0) return false;
    }
    return true;
  }

  static double average(List nums) =>
    nums.reduce((a,b)=>a+b)/nums.length;
}

// MAIN

  List list = [
    Programmer(),
    Dokter(),
    Guru()
  ];

  for (var p in list) {
    p.bekerja();
  }

  print(MathUtils.factorial(5));
  print(MathUtils.isPrime(7));
}

// Abstract class
abstract class MediaPlayer {
  void play();
  void stop();
}

// Interface Playable
abstract class Playable {
  void play();
}

// Interface Pausable
abstract class Pausable {
  void pause();
}

// Class AudioPlayer
class AudioPlayer extends MediaPlayer implements Playable, Pausable {
  @override
  void play() {
    print("Audio diputar");
  }

  @override
  void pause() {
    print("Audio dijeda");
  }

  @override
  void stop() {
    print("Audio dihentikan");
  }
}

// Class VideoPlayer
class VideoPlayer extends MediaPlayer implements Playable, Pausable {
  @override
  void play() {
    print("Video diputar");
  }

  @override
  void pause() {
    print("Video dijeda");
  }

  @override
  void stop() {
    print("Video dihentikan");
  }
}

// Main
void main() {
  MediaPlayer audio = AudioPlayer();
  MediaPlayer video = VideoPlayer();

  audio.play();
  audio.stop();

  video.play();
  video.stop();
}
